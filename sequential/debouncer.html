Sequential Input Conditioning

Debouncer

Filters out rapid transitions (bounce) from mechanical switches and buttons. Produces a clean, stable output signal.

Problem

Switch Bounce

Bounce Time

5-50 ms

Method

Counter / SR

Use Case

Buttons, Switches

Switch Bounce

#

When a mechanical switch closes, the contacts physically bounce, creating rapid on-off transitions before settling. A fast digital system sees these as multiple presses!

Why it matters: At 100 MHz, a 10ms bounce can create 1 million false edges! Counter-based debouncing is essential for reliable button input.

Debounce Methods

#
Counter-Based

Count consecutive samples at same level. Only change output when count reaches threshold.

if (input == output) count = 0 else count++ if (count >= THRESHOLD) output = input
Shift Register

Sample input into shift register. Output high only when all bits are 1 (or all 0).

shift = {shift[N-2:0], in} out_high = &shift // all 1s out_low = ~|shift // all 0s
Integrator

Count up when input high, down when low. Output changes at thresholds. Bidirectional counting.

Two-Flip-Flop + Timer

Synchronize input, then wait fixed time after any edge before sampling again.

RTL (Verilog / VHDL)

#
Verilog debouncer.v 
// Counter-based debouncer
module debouncer #(
    parameter COUNTER_BITS = 20  // ~10ms @ 100MHz
)(
    input  wire clk,
    input  wire rst_n,
    input  wire btn_in,          // Raw button input
    output reg  btn_out,         // Debounced output
    output wire btn_posedge,     // Rising edge pulse
    output wire btn_negedge      // Falling edge pulse
);

    reg [COUNTER_BITS-1:0] count;
    reg                    btn_sync1, btn_sync2;
    reg                    btn_prev;

    // Synchronizer (2 FF)
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            btn_sync1 <= 1'b0;
            btn_sync2 <= 1'b0;
        end else begin
            btn_sync1 <= btn_in;
            btn_sync2 <= btn_sync1;
        end
    end

    // Debounce counter
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            count   <= 0;
            btn_out <= 1'b0;
        end else begin
            if (btn_sync2 == btn_out) begin
                // Input matches output - reset counter
                count <= 0;
            end else begin
                // Input differs - count up
                if (count == {COUNTER_BITS{1'b1}}) begin
                    btn_out <= btn_sync2;
                    count   <= 0;
                end else begin
                    count <= count + 1;
                end
            end
        end
    end

    // Edge detection
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n)
            btn_prev <= 1'b0;
        else
            btn_prev <= btn_out;
    end

    assign btn_posedge = btn_out & ~btn_prev;
    assign btn_negedge = ~btn_out & btn_prev;

endmodule

// Shift register debouncer (alternative)
module debouncer_shift #(
    parameter STAGES = 8,
    parameter PRESCALE = 20  // Slow sample rate
)(
    input  wire clk,
    input  wire rst_n,
    input  wire btn_in,
    output reg  btn_out
);

    reg [PRESCALE-1:0]  prescaler;
    reg [STAGES-1:0]    shift;
    wire                sample_en;

    // Prescaler for slow sampling
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n)
            prescaler <= 0;
        else
            prescaler <= prescaler + 1;
    end
    
    assign sample_en = (prescaler == 0);

    // Shift register
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            shift   <= 0;
            btn_out <= 1'b0;
        end else if (sample_en) begin
            shift <= {shift[STAGES-2:0], btn_in};
            if (&shift)        // All 1s
                btn_out <= 1'b1;
            else if (~|shift)  // All 0s
                btn_out <= 1'b0;
        end
    end

endmodule
VHDL debouncer.vhd 
-- Counter-based debouncer
library IEEE;
use IEEE.STD_LOGIC_1164.all;
use IEEE.NUMERIC_STD.all;

entity debouncer is
    generic (
        COUNTER_MAX : integer := 1000000  -- ~10ms @ 100MHz
    );
    port (
        clk         : in  std_logic;
        rst_n       : in  std_logic;
        btn_in      : in  std_logic;
        btn_out     : out std_logic;
        btn_posedge : out std_logic;
        btn_negedge : out std_logic
    );
end debouncer;

architecture rtl of debouncer is
    signal count     : integer range 0 to COUNTER_MAX;
    signal sync1     : std_logic;
    signal sync2     : std_logic;
    signal debounced : std_logic;
    signal prev      : std_logic;
begin

    -- Synchronizer
    process(clk, rst_n)
    begin
        if rst_n = '0' then
            sync1 <= '0';
            sync2 <= '0';
        elsif rising_edge(clk) then
            sync1 <= btn_in;
            sync2 <= sync1;
        end if;
    end process;

    -- Debounce counter
    process(clk, rst_n)
    begin
        if rst_n = '0' then
            count <= 0;
            debounced <= '0';
        elsif rising_edge(clk) then
            if sync2 = debounced then
                count <= 0;
            else
                if count = COUNTER_MAX then
                    debounced <= sync2;
                    count <= 0;
                else
                    count <= count + 1;
                end if;
            end if;
        end if;
    end process;

    -- Edge detection
    process(clk, rst_n)
    begin
        if rst_n = '0' then
            prev <= '0';
        elsif rising_edge(clk) then
            prev <= debounced;
        end if;
    end process;

    btn_out <= debounced;
    btn_posedge <= debounced and (not prev);
    btn_negedge <= (not debounced) and prev;

end rtl;

Interview Notes

#
  • Why synchronize first? — Button is asynchronous to clock. Must pass through 2-FF synchronizer to avoid metastability before debounce logic.
  • How to size the counter? — Typical bounce is 5-50ms. Counter should be > worst-case bounce time. At 100MHz: 20-bit counter ≈ 10ms.
  • Shift register vs counter? — Shift register needs fewer bits but samples at fixed rate. Counter is more flexible but needs more bits.
  • Can debouncer handle glitches? — Yes, any pulse shorter than debounce time is filtered. But primary purpose is switch bounce.
  • What about release bounce? — Same mechanism. Debouncer handles both press and release transitions identically.
