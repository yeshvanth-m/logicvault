Sequential Clock Management

Frequency Divider

Divides input clock frequency by integer factor. Produces lower frequency output clock from higher frequency source.

Input

fin

Output

fin / N

Method

Counter

Use Case

Clock Gen, Timing

Power-of-2 Divider

#

Simplest divider: a counter's MSB divides by 2, next bit by 4, etc. Just use the appropriate counter bit as output.

Arbitrary Divider

#

For divide-by-N: count from 0 to N-1, toggle output at count 0. Or count from N-1 down to 0, toggle at 0.

Even Division (N even)

Count to N/2 - 1, toggle output, repeat. Gives 50% duty cycle.

// Divide by 6 if (cnt == 2) begin cnt <= 0; clk_out <= ~clk_out; end else cnt <= cnt + 1;
Non-50% Duty Cycle

Count full N cycles, pulse for 1 cycle. Duty cycle = 1/N.

// Divide by 5, 20% duty if (cnt == 4) cnt <= 0; else cnt <= cnt + 1; clk_out <= (cnt == 0);

Odd Division with 50% Duty

#

Odd division (÷3, ÷5, ÷7) with 50% duty cycle requires dual-edge technique: generate two clocks offset by half-cycle, then OR or XOR them.

Glitch Risk: ORing/XORing clocks from different edges can create glitches. For clean clocks, use PLL/MMCM or clock enables instead of divided clocks.

RTL (Verilog / VHDL)

#
Verilog freq_div.v 
// Power-of-2 divider (simple counter)
module freq_div_pow2 #(
    parameter DIV_BITS = 4  // Divides up to 2^16
)(
    input  wire               clk_in,
    input  wire               rst_n,
    output wire [DIV_BITS-1:0] clk_div  // [0]=÷2, [1]=÷4, etc.
);

    reg [DIV_BITS-1:0] cnt;

    always @(posedge clk_in or negedge rst_n) begin
        if (!rst_n)
            cnt <= 0;
        else
            cnt <= cnt + 1;
    end

    assign clk_div = cnt;

endmodule

// Arbitrary even divider (50% duty)
module freq_div_even #(
    parameter DIV = 10  // Must be even
)(
    input  wire clk_in,
    input  wire rst_n,
    output reg  clk_out
);

    localparam HALF = DIV / 2;
    localparam CNT_W = $clog2(HALF);
    
    reg [CNT_W-1:0] cnt;

    always @(posedge clk_in or negedge rst_n) begin
        if (!rst_n) begin
            cnt     <= 0;
            clk_out <= 1'b0;
        end else begin
            if (cnt == HALF - 1) begin
                cnt     <= 0;
                clk_out <= ~clk_out;
            end else begin
                cnt <= cnt + 1;
            end
        end
    end

endmodule

// Odd divider with 50% duty (dual-edge)
module freq_div_odd #(
    parameter DIV = 3  // Must be odd ≥ 3
)(
    input  wire clk_in,
    input  wire rst_n,
    output wire clk_out
);

    localparam CNT_W = $clog2(DIV);
    
    reg [CNT_W-1:0] cnt_p, cnt_n;
    reg            clk_p, clk_n;

    // Posedge counter
    always @(posedge clk_in or negedge rst_n) begin
        if (!rst_n) begin
            cnt_p <= 0;
            clk_p <= 1'b0;
        end else begin
            if (cnt_p == DIV - 1)
                cnt_p <= 0;
            else
                cnt_p <= cnt_p + 1;
            
            // High for first (DIV+1)/2 cycles
            clk_p <= (cnt_p < (DIV + 1) / 2);
        end
    end

    // Negedge counter (offset by half cycle)
    always @(negedge clk_in or negedge rst_n) begin
        if (!rst_n) begin
            cnt_n <= 0;
            clk_n <= 1'b0;
        end else begin
            if (cnt_n == DIV - 1)
                cnt_n <= 0;
            else
                cnt_n <= cnt_n + 1;
            
            clk_n <= (cnt_n < (DIV + 1) / 2);
        end
    end

    // Combine for 50% duty
    assign clk_out = clk_p | clk_n;

endmodule

// Clock enable (preferred in FPGAs)
module clk_enable #(
    parameter DIV = 10
)(
    input  wire clk,
    input  wire rst_n,
    output reg  ce    // Clock enable pulse
);

    localparam CNT_W = $clog2(DIV);
    reg [CNT_W-1:0] cnt;

    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            cnt <= 0;
            ce  <= 1'b0;
        end else begin
            if (cnt == DIV - 1) begin
                cnt <= 0;
                ce  <= 1'b1;
            end else begin
                cnt <= cnt + 1;
                ce  <= 1'b0;
            end
        end
    end

endmodule
VHDL freq_div.vhd 
-- Even frequency divider
library IEEE;
use IEEE.STD_LOGIC_1164.all;
use IEEE.NUMERIC_STD.all;

entity freq_div_even is
    generic (DIV : integer := 10);
    port (
        clk_in  : in  std_logic;
        rst_n   : in  std_logic;
        clk_out : out std_logic
    );
end freq_div_even;

architecture rtl of freq_div_even is
    constant HALF : integer := DIV / 2;
    signal cnt : integer range 0 to HALF - 1;
    signal clk_r : std_logic;
begin

    process(clk_in, rst_n)
    begin
        if rst_n = '0' then
            cnt <= 0;
            clk_r <= '0';
        elsif rising_edge(clk_in) then
            if cnt = HALF - 1 then
                cnt <= 0;
                clk_r <= not clk_r;
            else
                cnt <= cnt + 1;
            end if;
        end if;
    end process;

    clk_out <= clk_r;

end rtl;

-- Clock enable generator
entity clk_enable is
    generic (DIV : integer := 10);
    port (
        clk   : in  std_logic;
        rst_n : in  std_logic;
        ce    : out std_logic
    );
end clk_enable;

architecture rtl of clk_enable is
    signal cnt : integer range 0 to DIV - 1;
begin

    process(clk, rst_n)
    begin
        if rst_n = '0' then
            cnt <= 0;
            ce <= '0';
        elsif rising_edge(clk) then
            if cnt = DIV - 1 then
                cnt <= 0;
                ce <= '1';
            else
                cnt <= cnt + 1;
                ce <= '0';
            end if;
        end if;
    end process;

end rtl;

Interview Notes

#
  • Clock enable vs divided clock? — Clock enable is safer: single clock domain, no skew issues. Divided clock creates new domain requiring synchronization.
  • How to divide by odd number with 50% duty? — Use dual-edge technique: generate clocks on both edges, combine them. Or use DLL/PLL.
  • Why does MSB divide by 2? — Binary counter: MSB toggles every 2^(N-1) counts. For N-bit counter, bit[0] toggles every clock (÷2).
  • Fractional division? — Use DDS (Direct Digital Synthesis) or fractional-N PLL. Cannot do exact fractional with simple counters.
  • Glitch-free clock switching? — Synchronize switch signal to both clocks, gate each clock, AND clocks with synced enables.
