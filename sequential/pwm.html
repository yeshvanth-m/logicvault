Sequential Modulation

PWM Generator

Pulse Width Modulation — varies the duty cycle of a periodic signal to encode information or control average power.

Parameter

Duty Cycle

Range

0% - 100%

Resolution

N-bit counter

Use Case

Motor, LED, DAC

PWM Principle

#

Compare a counter value to a threshold. Output is HIGH when counter < threshold, LOW otherwise. Varying threshold changes duty cycle.

Resolution

N-bit counter gives 2N duty cycle levels. 8-bit = 256 levels, 10-bit = 1024 levels.

Frequency

fpwm = fclk / 2N. Higher resolution = lower PWM frequency.

PWM Types

#
Edge-Aligned (Sawtooth)

Counter counts up, resets at max. PWM edge aligned to period start. Simple, common.

▁▂▃▄▅▆▇█▁▂▃▄▅▆▇█ ────┐ ────┐
Center-Aligned (Triangle)

Counter counts up then down. PWM centered in period. Symmetric, less harmonics.

▁▂▃▄▅▆▇█▇▆▅▄▃▂▁ ──────┐ ────────┐
Phase-Correct

Like center-aligned but maintains phase relationship across duty changes. Motor control.

Complementary

Two outputs, one inverted. Dead-time insertion prevents shoot-through. H-bridge drivers.

RTL (Verilog / VHDL)

#
Verilog pwm.v 
// Edge-aligned PWM (basic)
module pwm_basic #(
    parameter WIDTH = 8  // 8-bit = 256 levels
)(
    input  wire             clk,
    input  wire             rst_n,
    input  wire [WIDTH-1:0] duty,    // 0=0%, 255=100%
    output reg              pwm_out
);

    reg [WIDTH-1:0] counter;

    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            counter <= 0;
            pwm_out <= 1'b0;
        end else begin
            counter <= counter + 1;  // Wraps at 2^WIDTH
            pwm_out <= (counter < duty);
        end
    end

endmodule

// Configurable period PWM
module pwm_config #(
    parameter WIDTH = 16
)(
    input  wire             clk,
    input  wire             rst_n,
    input  wire [WIDTH-1:0] period,  // Period in clocks
    input  wire [WIDTH-1:0] duty,    // Duty in clocks
    output reg              pwm_out
);

    reg [WIDTH-1:0] counter;

    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            counter <= 0;
            pwm_out <= 1'b0;
        end else begin
            if (counter >= period - 1)
                counter <= 0;
            else
                counter <= counter + 1;
            
            pwm_out <= (counter < duty);
        end
    end

endmodule

// Center-aligned PWM
module pwm_center #(
    parameter WIDTH = 8
)(
    input  wire             clk,
    input  wire             rst_n,
    input  wire [WIDTH-1:0] duty,
    output reg              pwm_out
);

    reg [WIDTH-1:0] counter;
    reg            dir;  // 0=up, 1=down

    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            counter <= 0;
            dir     <= 1'b0;
            pwm_out <= 1'b0;
        end else begin
            // Triangle counter
            if (dir == 1'b0) begin  // Counting up
                if (counter == {WIDTH{1'b1}})
                    dir <= 1'b1;
                else
                    counter <= counter + 1;
            end else begin           // Counting down
                if (counter == 0)
                    dir <= 1'b0;
                else
                    counter <= counter - 1;
            end
            
            pwm_out <= (counter < duty);
        end
    end

endmodule

// Complementary PWM with dead-time
module pwm_comp #(
    parameter WIDTH = 8,
    parameter DEAD_TIME = 4  // Clocks of dead-time
)(
    input  wire             clk,
    input  wire             rst_n,
    input  wire [WIDTH-1:0] duty,
    output wire             pwm_h,   // High-side
    output wire             pwm_l    // Low-side (complementary)
);

    reg [WIDTH-1:0] counter;
    wire           pwm_raw;
    reg [7:0]      dead_cnt;
    reg            pwm_h_r, pwm_l_r;
    reg            pwm_prev;

    always @(posedge clk or negedge rst_n) begin
        if (!rst_n)
            counter <= 0;
        else
            counter <= counter + 1;
    end
    
    assign pwm_raw = (counter < duty);

    // Dead-time insertion
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            pwm_h_r  <= 1'b0;
            pwm_l_r  <= 1'b0;
            pwm_prev <= 1'b0;
            dead_cnt <= 0;
        end else begin
            pwm_prev <= pwm_raw;
            
            if (pwm_raw != pwm_prev) begin
                // Edge detected - start dead-time
                dead_cnt <= DEAD_TIME;
                pwm_h_r  <= 1'b0;
                pwm_l_r  <= 1'b0;
            end else if (dead_cnt != 0) begin
                dead_cnt <= dead_cnt - 1;
            end else begin
                pwm_h_r <= pwm_raw;
                pwm_l_r <= ~pwm_raw;
            end
        end
    end

    assign pwm_h = pwm_h_r;
    assign pwm_l = pwm_l_r;

endmodule
VHDL pwm.vhd 
-- Basic PWM generator
library IEEE;
use IEEE.STD_LOGIC_1164.all;
use IEEE.NUMERIC_STD.all;

entity pwm_basic is
    generic (WIDTH : integer := 8);
    port (
        clk     : in  std_logic;
        rst_n   : in  std_logic;
        duty    : in  std_logic_vector(WIDTH-1 downto 0);
        pwm_out : out std_logic
    );
end pwm_basic;

architecture rtl of pwm_basic is
    signal counter : unsigned(WIDTH-1 downto 0);
begin

    process(clk, rst_n)
    begin
        if rst_n = '0' then
            counter <= (others => '0');
            pwm_out <= '0';
        elsif rising_edge(clk) then
            counter <= counter + 1;
            if counter < unsigned(duty) then
                pwm_out <= '1';
            else
                pwm_out <= '0';
            end if;
        end if;
    end process;

end rtl;

-- Configurable period PWM
entity pwm_config is
    generic (WIDTH : integer := 16);
    port (
        clk     : in  std_logic;
        rst_n   : in  std_logic;
        period  : in  std_logic_vector(WIDTH-1 downto 0);
        duty    : in  std_logic_vector(WIDTH-1 downto 0);
        pwm_out : out std_logic
    );
end pwm_config;

architecture rtl of pwm_config is
    signal counter : unsigned(WIDTH-1 downto 0);
begin

    process(clk, rst_n)
    begin
        if rst_n = '0' then
            counter <= (others => '0');
            pwm_out <= '0';
        elsif rising_edge(clk) then
            if counter >= unsigned(period) - 1 then
                counter <= (others => '0');
            else
                counter <= counter + 1;
            end if;
            
            if counter < unsigned(duty) then
                pwm_out <= '1';
            else
                pwm_out <= '0';
            end if;
        end if;
    end process;

end rtl;

Interview Notes

#
  • Why use PWM for analog control? — Average voltage = Vdd × duty. With LP filter, PWM becomes poor-man's DAC. Digital control of analog quantities.
  • Resolution vs frequency trade-off? — Higher resolution = more counter bits = lower PWM frequency. fpwm = fclk / 2N.
  • What is dead-time? — Delay between turning off one switch and turning on complementary switch. Prevents both being on simultaneously (shoot-through).
  • Center-aligned vs edge-aligned? — Center-aligned produces less high-frequency harmonics (smoother). Important for motor drives to reduce noise.
  • How does PWM drive a motor? — PWM to H-bridge. Average voltage proportional to duty. Low-pass filtering by motor inductance smooths the drive.
