Mealy Machine

A finite state machine where outputs depend on both the current state and the current inputs. Outputs can change asynchronously with input changes, enabling faster response but requiring careful timing analysis.

State + Input

Structure

#

Mealy Machine Definition

Next State = f(Current State, Input)
Output = g(Current State, Input)

State Diagram Notation

#

In Mealy diagrams, outputs are labeled on transitions (edges), not states. Format: input / output

State Table

#

Example: Edge detector (outputs 1 when input transitions from 0 to 1)

Current State Input Next State Output
S0 (was 0)0S00
S0 (was 0)1S11
S1 (was 1)0S00
S1 (was 1)1S10

Output is 1 only on the transition from S0→S1 (rising edge). The output responds immediately when input changes — no clock cycle delay.

Timing Behavior

#

Glitch risk: Since Mealy outputs change immediately with inputs, they can produce glitches if inputs have propagation skew. Register outputs if driving other synchronous logic.

RTL (Verilog / VHDL)

#
Verilog mealy_edge_detector.v 
// Mealy FSM: Rising Edge Detector
// Output goes high immediately when input rises
module mealy_edge_detector (
    input  wire clk,
    input  wire rst_n,
    input  wire din,
    output wire edge_detected  // Combinational output
);

    // State encoding
    localparam S0 = 1'b0;  // Previous input was 0
    localparam S1 = 1'b1;  // Previous input was 1

    reg state, next_state;

    // State register
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n)
            state <= S0;
        else
            state <= next_state;
    end

    // Next state logic (combinational)
    always @(*) begin
        case (state)
            S0: next_state = din ? S1 : S0;
            S1: next_state = din ? S1 : S0;
            default: next_state = S0;
        endcase
    end

    // Mealy output: depends on state AND input
    assign edge_detected = (state == S0) && din;

endmodule

// Mealy FSM with registered output (glitch-free)
module mealy_registered (
    input  wire clk,
    input  wire rst_n,
    input  wire din,
    output reg  edge_out  // Registered (1 cycle delay)
);

    reg state;
    localparam S0 = 1'b0, S1 = 1'b1;

    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            state    <= S0;
            edge_out <= 1'b0;
        end else begin
            // Compute Mealy output and register it
            edge_out <= (state == S0) && din;
            // State transition
            state <= din ? S1 : S0;
        end
    end

endmodule
VHDL mealy_edge_detector.vhd 
-- Mealy FSM: Rising Edge Detector
library IEEE;
use IEEE.STD_LOGIC_1164.all;

entity mealy_edge_detector is
    port (
        clk           : in  std_logic;
        rst_n         : in  std_logic;
        din           : in  std_logic;
        edge_detected : out std_logic
    );
end mealy_edge_detector;

architecture behavioral of mealy_edge_detector is
    type state_type is (S0, S1);
    signal state, next_state : state_type;
begin

    -- State register
    process(clk, rst_n)
    begin
        if rst_n = '0' then
            state <= S0;
        elsif rising_edge(clk) then
            state <= next_state;
        end if;
    end process;

    -- Next state logic
    process(state, din)
    begin
        case state is
            when S0 =>
                if din = '1' then
                    next_state <= S1;
                else
                    next_state <= S0;
                end if;
            when S1 =>
                if din = '1' then
                    next_state <= S1;
                else
                    next_state <= S0;
                end if;
        end case;
    end process;

    -- Mealy output: depends on state AND input
    edge_detected <= '1' when (state = S0 and din = '1') else '0';

end behavioral;
