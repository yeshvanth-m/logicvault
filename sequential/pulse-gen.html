Pulse Generator

Generates fixed-width output pulses from input edges. Used for triggering, timing, and signal conditioning.

Input

Edge / Level

Output

Fixed-Width Pulse

Control

Width Counter

Use Case

Trigger, Strobe

Pulse Types

#
Single-Cycle Pulse

One clock cycle wide. Generated from edge detection. Most common for synchronous triggers.

Multi-Cycle Pulse

Configurable width via counter. Used when downstream needs longer assertion time.

Retriggerable

New trigger resets width counter. Output stays high while triggers keep coming.

Non-Retriggerable

Ignores triggers while output is active. Fixed pulse regardless of input activity.

One-Shot (Monostable)

#

Use Cases: Generate write strobes, trigger ADC conversions, create fixed-width reset pulses, debounced button edge pulses.

Edge Detection

#

Basic building block for pulse generation. Compares current and previous sample to detect transitions.

RTL (Verilog / VHDL)

#
Verilog pulse_gen.v 
// Single-cycle edge detector
module edge_detect (
    input  wire clk,
    input  wire rst_n,
    input  wire sig_in,
    output wire posedge_out,
    output wire negedge_out,
    output wire anyedge_out
);

    reg sig_prev;

    always @(posedge clk or negedge rst_n) begin
        if (!rst_n)
            sig_prev <= 1'b0;
        else
            sig_prev <= sig_in;
    end

    assign posedge_out = sig_in & ~sig_prev;
    assign negedge_out = ~sig_in & sig_prev;
    assign anyedge_out = sig_in ^ sig_prev;

endmodule

// Multi-cycle pulse generator (one-shot)
module pulse_gen #(
    parameter WIDTH_BITS = 8   // Max pulse width = 2^8 = 256 cycles
)(
    input  wire                 clk,
    input  wire                 rst_n,
    input  wire                 trigger,
    input  wire [WIDTH_BITS-1:0] pulse_width,  // 0 = 1 cycle
    output reg                  pulse_out,
    output wire                 busy
);

    reg [WIDTH_BITS-1:0] count;
    reg                  active;
    reg                  trig_prev;
    wire                 trig_edge;

    assign trig_edge = trigger & ~trig_prev;
    assign busy = active;

    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            trig_prev <= 1'b0;
            count     <= 0;
            active    <= 1'b0;
            pulse_out <= 1'b0;
        end else begin
            trig_prev <= trigger;
            
            if (active) begin
                if (count == 0) begin
                    active    <= 1'b0;
                    pulse_out <= 1'b0;
                end else begin
                    count <= count - 1;
                end
            end else if (trig_edge) begin
                active    <= 1'b1;
                pulse_out <= 1'b1;
                count     <= pulse_width;
            end
        end
    end

endmodule

// Retriggerable pulse generator
module pulse_gen_retrig #(
    parameter WIDTH_BITS = 8
)(
    input  wire                 clk,
    input  wire                 rst_n,
    input  wire                 trigger,
    input  wire [WIDTH_BITS-1:0] pulse_width,
    output reg                  pulse_out
);

    reg [WIDTH_BITS-1:0] count;
    reg                  trig_prev;
    wire                 trig_edge;

    assign trig_edge = trigger & ~trig_prev;

    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            trig_prev <= 1'b0;
            count     <= 0;
            pulse_out <= 1'b0;
        end else begin
            trig_prev <= trigger;
            
            if (trig_edge) begin
                // Reset counter on any trigger
                count     <= pulse_width;
                pulse_out <= 1'b1;
            end else if (count != 0) begin
                count <= count - 1;
                if (count == 1)
                    pulse_out <= 1'b0;
            end
        end
    end

endmodule
VHDL pulse_gen.vhd 
-- Edge detector
library IEEE;
use IEEE.STD_LOGIC_1164.all;

entity edge_detect is
    port (
        clk         : in  std_logic;
        rst_n       : in  std_logic;
        sig_in      : in  std_logic;
        posedge_out : out std_logic;
        negedge_out : out std_logic
    );
end edge_detect;

architecture rtl of edge_detect is
    signal prev : std_logic;
begin
    process(clk, rst_n)
    begin
        if rst_n = '0' then
            prev <= '0';
        elsif rising_edge(clk) then
            prev <= sig_in;
        end if;
    end process;

    posedge_out <= sig_in and (not prev);
    negedge_out <= (not sig_in) and prev;
end rtl;

-- Multi-cycle pulse generator
entity pulse_gen is
    generic (WIDTH_BITS : integer := 8);
    port (
        clk         : in  std_logic;
        rst_n       : in  std_logic;
        trigger     : in  std_logic;
        pulse_width : in  std_logic_vector(WIDTH_BITS-1 downto 0);
        pulse_out   : out std_logic;
        busy        : out std_logic
    );
end pulse_gen;

architecture rtl of pulse_gen is
    signal count     : unsigned(WIDTH_BITS-1 downto 0);
    signal active    : std_logic;
    signal trig_prev : std_logic;
    signal trig_edge : std_logic;
begin
    trig_edge <= trigger and (not trig_prev);
    busy <= active;

    process(clk, rst_n)
    begin
        if rst_n = '0' then
            trig_prev <= '0';
            count <= (others => '0');
            active <= '0';
            pulse_out <= '0';
        elsif rising_edge(clk) then
            trig_prev <= trigger;
            if active = '1' then
                if count = 0 then
                    active <= '0';
                    pulse_out <= '0';
                else
                    count <= count - 1;
                end if;
            elsif trig_edge = '1' then
                active <= '1';
                pulse_out <= '1';
                count <= unsigned(pulse_width);
            end if;
        end if;
    end process;
end rtl;

Interview Notes

#
  • Why use edge detection? — Converts level changes to single-cycle pulses. Essential for triggering single events from button presses or status changes.
  • Retriggerable vs non-retriggerable? — Retriggerable resets timer on each trigger (extends pulse). Non-retriggerable ignores triggers while active (fixed width).
  • How to stretch a pulse? — Use counter that loads on trigger edge, decrements each cycle, output high while count > 0.
  • What about input glitches? — Edge detector will produce spurious pulses. Consider synchronizer + debouncer before edge detection.
  • Cross-clock pulse? — Use pulse synchronizer: stretch pulse in source domain, synchronize, then edge detect in destination domain.
