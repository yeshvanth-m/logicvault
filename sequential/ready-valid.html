Sequential Interface Protocol

Ready/Valid Handshake

Standard flow control protocol for transferring data between producer and consumer. Handles backpressure and ensures reliable data transfer.

Signals

valid, ready, data

Transfer

valid & ready

Backpressure

ready = 0

Standards

AXI, AXI-Stream

Protocol Rules

#
Producer Rules
  • Assert valid when data is available
  • Hold data stable while valid=1
  • Keep valid=1 until ready=1 seen
  • valid must NOT depend on ready
Consumer Rules
  • Assert ready when able to accept
  • Sample data when both signals high
  • ready MAY depend on valid
  • Can deassert ready for backpressure

Critical: Valid must NOT wait for ready before asserting! This prevents deadlock where both sides wait for the other.

Verilog ready_valid.v 
// Simple producer (source)
module producer #(
    parameter WIDTH = 8
)(
    input  wire             clk,
    input  wire             rst_n,
    input  wire             start,       // External trigger
    input  wire [WIDTH-1:0] data_in,     // Data to send
    
    // Ready/Valid interface (output)
    output reg              out_valid,
    input  wire             out_ready,
    output reg  [WIDTH-1:0] out_data
);

    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            out_valid <= 1'b0;
            out_data  <= 0;
        end else begin
            if (out_valid && out_ready) begin
                // Transfer complete - can load new or go idle
                out_valid <= start;  // Immediate next if start
                out_data  <= data_in;
            end else if (!out_valid && start) begin
                // New data to send
                out_valid <= 1'b1;
                out_data  <= data_in;
            end
            // Note: holds valid until ready seen
        end
    end

endmodule

// Simple consumer (sink)
module consumer #(
    parameter WIDTH = 8
)(
    input  wire             clk,
    input  wire             rst_n,
    
    // Ready/Valid interface (input)
    input  wire             in_valid,
    output wire             in_ready,
    input  wire [WIDTH-1:0] in_data,
    
    // Output to downstream
    output reg  [WIDTH-1:0] data_out,
    output reg              data_valid
);

    reg busy;
    
    // Ready when not busy
    assign in_ready = !busy;

    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            busy       <= 1'b0;
            data_out   <= 0;
            data_valid <= 1'b0;
        end else begin
            data_valid <= 1'b0;  // Default
            
            if (in_valid && in_ready) begin
                // Accept data
                data_out   <= in_data;
                data_valid <= 1'b1;
                busy       <= 1'b1;  // Simulate processing
            end
            
            if (busy)
                busy <= 1'b0;  // One cycle "busy"
        end
    end

endmodule

// Pipeline register with ready/valid
module pipe_stage_rv #(
    parameter WIDTH = 8
)(
    input  wire             clk,
    input  wire             rst_n,
    
    // Upstream (input)
    input  wire             in_valid,
    output wire             in_ready,
    input  wire [WIDTH-1:0] in_data,
    
    // Downstream (output)
    output reg              out_valid,
    input  wire             out_ready,
    output reg  [WIDTH-1:0] out_data
);

    // Ready if downstream ready OR we have no data
    assign in_ready = out_ready | ~out_valid;

    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            out_valid <= 1'b0;
            out_data  <= 0;
        end else if (in_ready) begin
            // Can accept: pass through or clear
            out_valid <= in_valid;
            out_data  <= in_data;
        end
    end

endmodule

// Check for proper handshake (simulation only)
module rv_checker #(
    parameter WIDTH = 8
)(
    input wire             clk,
    input wire             valid,
    input wire             ready,
    input wire [WIDTH-1:0] data
);

    reg            prev_valid;
    reg [WIDTH-1:0] prev_data;

    always @(posedge clk) begin
        prev_valid <= valid;
        prev_data  <= data;
        
        // Check: valid must not drop without ready
        if (prev_valid && !valid && !ready) begin
            $error("Valid dropped without handshake!");
        end
        
        // Check: data must be stable while valid
        if (prev_valid && valid && !ready) begin
            if (prev_data !== data) begin
                $error("Data changed while waiting for ready!");
            end
        end
    end

endmodule
VHDL ready_valid.vhd 
-- Pipeline stage with ready/valid
library IEEE;
use IEEE.STD_LOGIC_1164.all;

entity pipe_stage_rv is
    generic (WIDTH : integer := 8);
    port (
        clk       : in  std_logic;
        rst_n     : in  std_logic;
        -- Upstream
        in_valid  : in  std_logic;
        in_ready  : out std_logic;
        in_data   : in  std_logic_vector(WIDTH-1 downto 0);
        -- Downstream
        out_valid : out std_logic;
        out_ready : in  std_logic;
        out_data  : out std_logic_vector(WIDTH-1 downto 0)
    );
end pipe_stage_rv;

architecture rtl of pipe_stage_rv is
    signal valid_r  : std_logic;
    signal data_r   : std_logic_vector(WIDTH-1 downto 0);
    signal ready_i  : std_logic;
begin

    ready_i <= out_ready or (not valid_r);
    in_ready <= ready_i;
    out_valid <= valid_r;
    out_data <= data_r;

    process(clk, rst_n)
    begin
        if rst_n = '0' then
            valid_r <= '0';
            data_r <= (others => '0');
        elsif rising_edge(clk) then
            if ready_i = '1' then
                valid_r <= in_valid;
                data_r <= in_data;
            end if;
        end if;
    end process;

end rtl;

Interview Notes

#
  • Why can't valid wait for ready? — Creates deadlock! Both sides would wait forever. Producer must assert valid independently of ready.
  • What is backpressure? — Consumer asserting ready=0 to pause the producer. Data stays valid; producer waits.
  • Bubble vs backpressure? — Bubble = valid=0 (no data). Backpressure = ready=0 (consumer full). Both stall pipeline but different causes.
  • What's a skid buffer? — Extra register to maintain throughput during backpressure. Stores one item when downstream stalls.
  • How to verify handshake? — Check valid doesn't drop without ready, data stable while waiting, no X/Z on control signals.
  • AXI vs AXI-Stream? — AXI: memory-mapped with address/data channels. AXI-Stream: streaming data, no address, simpler.
