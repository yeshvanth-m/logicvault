Routing Resource Sharing

Arbiter

Resolves contention when multiple requestors compete for a shared resource. Selects one winner based on priority, fairness, or other policies.

Function

Resource Selection

Inputs

N Request Lines

Output

Grant (one-hot)

Use Case

Bus, Cache, NoC

Fixed Priority Arbiter

#

Simplest arbiter: request 0 has highest priority, then 1, 2, etc. Low-indexed requests always win. Can cause starvation of lower-priority requestors.

Round Robin Arbiter

#

Fair arbitration: priority rotates after each grant. Last winner gets lowest priority next time. Ensures all requestors get service.

Implementation: Use a mask based on pointer position. Mask off lower-priority requests, run priority encoder. If no masked request, use unmasked priority encoder.

Weighted Round Robin

#

Each requestor gets a weight (number of slots). Higher weight = more grants per round. Used for QoS and bandwidth allocation.

Credit-Based

Each port has credit counter. Decremented on grant, reset when round completes. Skip port when credit = 0.

Deficit Round Robin

Unused credits carry over. Good for variable-size packets. Fair even with different burst sizes.

RTL (Verilog / VHDL)

#
Verilog arbiter.v 
// Fixed Priority Arbiter
module arbiter_fixed #(
    parameter N = 4
)(
    input  wire [N-1:0] req,
    output wire [N-1:0] gnt
);

    wire [N-1:0] higher_pri_req;
    
    // higher_pri_req[i] = any request with index < i
    assign higher_pri_req[0] = 1'b0;
    genvar i;
    generate
        for (i = 1; i < N; i = i + 1) begin : gen_hpr
            assign higher_pri_req[i] = |req[i-1:0];
        end
    endgenerate
    
    // Grant if requesting and no higher priority
    assign gnt = req & ~higher_pri_req;

endmodule

// Round Robin Arbiter
module arbiter_rr #(
    parameter N = 4
)(
    input  wire        clk,
    input  wire        rst_n,
    input  wire [N-1:0] req,
    output reg  [N-1:0] gnt
);

    reg [N-1:0] mask;  // Masks lower priority requests
    wire [N-1:0] masked_req;
    wire [N-1:0] mask_gnt;
    wire [N-1:0] unmask_gnt;
    wire        no_masked_req;

    // Mask requests below last grant
    assign masked_req = req & mask;
    
    // Fixed priority on masked requests
    arbiter_fixed #(N) masked_arb (
        .req(masked_req),
        .gnt(mask_gnt)
    );
    
    // Fixed priority on all requests (fallback)
    arbiter_fixed #(N) unmask_arb (
        .req(req),
        .gnt(unmask_gnt)
    );
    
    // Use masked if any, else unmasked
    assign no_masked_req = ~|masked_req;
    
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            gnt  <= {N{1'b0}};
            mask <= {N{1'b1}};  // All enabled
        end else begin
            if (|req) begin
                gnt <= no_masked_req ? unmask_gnt : mask_gnt;
                // Update mask: set bits above winner
                mask <= no_masked_req ? 
                    ({N{1'b1}} << $clog2(unmask_gnt+1)) :
                    ({N{1'b1}} << $clog2(mask_gnt+1));
            end else begin
                gnt <= {N{1'b0}};
            end
        end
    end

endmodule

// Cleaner Round Robin using rotate
module arbiter_rr_simple #(
    parameter N = 4
)(
    input  wire        clk,
    input  wire        rst_n,
    input  wire [N-1:0] req,
    output wire [N-1:0] gnt,
    output wire        valid
);

    reg [$clog2(N)-1:0] ptr;  // Points to highest priority
    wire [2*N-1:0] double_req;
    wire [2*N-1:0] double_gnt;
    
    // Double the request for wraparound
    assign double_req = {req, req};
    
    // Shift and mask
    wire [N-1:0] shifted_req = double_req[ptr +: N];
    wire [N-1:0] shifted_gnt;
    
    // Fixed priority on shifted
    arbiter_fixed #(N) pri_arb (
        .req(shifted_req),
        .gnt(shifted_gnt)
    );
    
    // Unshift grant
    assign double_gnt = {{N{1'b0}}, shifted_gnt} << ptr;
    assign gnt = double_gnt[N-1:0] | double_gnt[2*N-1:N];
    assign valid = |gnt;
    
    // Update pointer
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n)
            ptr <= 0;
        else if (valid)
            ptr <= (ptr + 1) % N;  // Move to next
    end

endmodule
VHDL arbiter.vhd 
-- Fixed Priority Arbiter
library IEEE;
use IEEE.STD_LOGIC_1164.all;
use IEEE.NUMERIC_STD.all;

entity arbiter_fixed is
    generic (N : integer := 4);
    port (
        req : in  std_logic_vector(N-1 downto 0);
        gnt : out std_logic_vector(N-1 downto 0)
    );
end arbiter_fixed;

architecture rtl of arbiter_fixed is
begin
    process(req)
        variable found : std_logic;
    begin
        found := '0';
        gnt <= (others => '0');
        for i in 0 to N-1 loop
            if req(i) = '1' and found = '0' then
                gnt(i) <= '1';
                found := '1';
            end if;
        end loop;
    end process;
end rtl;

-- Round Robin Arbiter
entity arbiter_rr is
    generic (N : integer := 4);
    port (
        clk   : in  std_logic;
        rst_n : in  std_logic;
        req   : in  std_logic_vector(N-1 downto 0);
        gnt   : out std_logic_vector(N-1 downto 0)
    );
end arbiter_rr;

architecture rtl of arbiter_rr is
    signal ptr : integer range 0 to N-1;
begin
    process(clk, rst_n)
        variable idx : integer;
    begin
        if rst_n = '0' then
            ptr <= 0;
            gnt <= (others => '0');
        elsif rising_edge(clk) then
            gnt <= (others => '0');
            for i in 0 to N-1 loop
                idx := (ptr + i) mod N;
                if req(idx) = '1' then
                    gnt(idx) <= '1';
                    ptr <= (idx + 1) mod N;
                    exit;
                end if;
            end loop;
        end if;
    end process;
end rtl;

Interview Notes

#
  • What is starvation? — A low-priority requestor never gets service because higher-priority ones are always active. Fixed priority is prone to this.
  • How does round robin prevent starvation? — Priority rotates after each grant. Even the lowest original priority becomes highest eventually.
  • What's the latency of a fixed priority arbiter? — Combinational: one gate delay per priority level. Can be O(log N) with parallel prefix network.
  • When would you use fixed priority? — Interrupt controllers (some interrupts are truly more urgent), or when starvation is acceptable.
  • How to handle simultaneous request and release? — Common to have req hold high during entire transaction. Grant is held until transaction completes.
  • Matrix arbiter vs round robin? — Matrix arbiter tracks pairwise priorities. More state but can handle complex scenarios. RR is simpler for equal weights.
